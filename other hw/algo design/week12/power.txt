    FUNCTION GetGreatestAverage(mainList):
        SET windowSize TO 10
        SET allAverages TO empty list
        
        // Ensure the list is long enough to have at least one subarray
        IF length of mainList < windowSize:
            RETURN "List too short"

        // Loop through the list using indexes
        // We stop at (Length - WindowSize) so we don't go out of bounds
        FOR i FROM 0 TO (length of mainList - windowSize):
            
            // 1. Grab the subarray using indexes (e.g., 0-9, 1-10, etc.)
            SET currentSubarray TO mainList from index i to i + windowSize
            
            // 2. Calculate the average of the currentSubarray
            SET currentSum TO 0
            FOR EACH number IN currentSubarray:
                currentSum = currentSum + number
            
            SET currentAverage TO currentSum / windowSize
            
            // 3. Store the average in a new list
            APPEND currentAverage TO allAverages

        // 4. Find and return the greatest average from our stored list
        SET maxAverage TO first element of allAverages
        FOR EACH avg IN allAverages:
            IF avg > maxAverage:
                maxAverage = avg
                
        RETURN maxAverage
    END FUNCTION
